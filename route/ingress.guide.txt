# Exact match — only /mypath (no trailing slash, nothing after)
oc create ingress myapp --rule="myapp.apps-crc.testing/mypath=myapp:80"

# Prefix match — /mypath, /mypath/, /mypath/anything
oc create ingress myapp --rule="myapp.apps-crc.testing/mypath*=myapp:80"

# Root prefix — matches everything (equivalent to pathType: Prefix on /)
oc create ingress myapp --rule="myapp.apps-crc.testing/*=myapp:80"

# Multiple rules in one command (different paths to different services)
oc create ingress myapp \
  --rule="myapp.apps-crc.testing/api*=myapp-api:8080" \
  --rule="myapp.apps-crc.testing/web*=myapp-web:3000"
